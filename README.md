interface ObjectLinkPhpProvider - провайдер между js и php objectlink
	invoke(funcName, funcParams, uid) //отправка http post запроса к php-скрипту провайдера objectlink (как gOrm, но не массив, а именованные параметры)

	gO(n) //получение массива объектов с именем n
	gO(n, cid) //получение oid объекта с именем n,  связанного с объектом/классом oid
	gL(oid1, oid2) //получение id связи двух связанных объектов oid1 и oid2, если они связаны	
	gN(oid) //получение имени объекта по oid
	cO(n) //поиск и создание (при отсутствии) объекта с именем n
	cO(n, cid) = cL(cO(n), cid) //поиск создание (при отсутствии) объекта с именем n и привязкой к объекту-классу cid
	cO(n, сid, pid) = cL(cO(n, cid), pid) //поиск создание (при отсутствии) объекта с именем n и привязкой к объекту-классу cid и объекту-родителю pid
	cL(oid1,oid2) //поиск и создание (при отсутствии) связи между объектами oid1 и oid2
	eO(oid) //удаление (обнуление) объекта oid и всех его связей
	eL(oid1, oid2)// удаление (обнуление) связи между объектами oid1 и oid2
	uO(oid, n, c, d, u) //поиск объекта по oid и обновление его параметров
	uL(oid1, oid1, c, d, u) //поиск связи между объектами oid1 и oid2 и обновление ее параметров
	
	gLogin({login, password}) //проверка подлиности логина, возврашает uid пользователя при успешной авторизации или null
	policy(uid, functionNames[]) //проверка доступности указанных имен функций functionNames[] для пользователя uid
	gT2({classNames[], classesLinks[][c1,c2],...})
	gAnd({oids[],...}) //возвращает массив объектов, связанных со всеми объектами, указанными в oids[]
	createPolygonObject({coordinats[], oid}) //создает объект класса полигон с координатами класса координаты по координатам coordinats[] и связывает объект полигона с объектом oid
	+getObjectLatLngs(oid) //возвращает массив объектов LatLng/PolyLine/Polygon, связанных с объектом oid
	//класс: LatLng -> объект: "[Lat, Lng]" или "Lat Lng"; класс: PolyLine -> объект: "[[Lat, Lng],[Lat, Lng]]; класс: Polygon -> объект: "[[Lat, Lng],[Lat, Lng]]""
